Astro frontmatter section (between the Astro fence markers):

```typescript
// Astro frontmatter (top of .astro file between fence markers)
export const prerender = false;
import { db } from '../../db/index';
import { landingPages } from '../../db/schema';
import { eq } from 'drizzle-orm';

const { slug } = Astro.params;
if (!slug) return Astro.redirect('/404');

const [page] = await db
  .select()
  .from(landingPages)
  .where(eq(landingPages.slug, slug))
  .limit(1);

if (!page) return Astro.redirect('/404');

// Parse social links from JSON string
const socialLinks: string[] = page.socialLinks ? JSON.parse(page.socialLinks) : [];

// Set cache headers — landing page content rarely changes mid-campaign
Astro.response.headers.set('Cache-Control', 'public, s-maxage=300, stale-while-revalidate=60');
```

HTML structure — full standalone page, no Layout.astro wrapper:
