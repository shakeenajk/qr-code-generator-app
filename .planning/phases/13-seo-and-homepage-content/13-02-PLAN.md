    ---
    // src/pages/use-cases/[slug].astro
    import Layout from '../../layouts/Layout.astro';
    import Header from '../../components/Header.astro';
    import Footer from '../../components/Footer.astro';
    import { SEO } from 'astro-seo';
    import { USE_CASES } from '../../data/useCases';

    export function getStaticPaths() {
      return USE_CASES.map((useCase) => ({
        params: { slug: useCase.slug },
        props: { useCase },
      }));
    }

    const { useCase } = Astro.props;

    const canonicalUrl = new URL(`/use-cases/${useCase.slug}/`, Astro.site).toString();
    const hubUrl = new URL('/use-cases/', Astro.site).toString();

    const breadcrumbSchema = {
      '@context': 'https://schema.org',
      '@type': 'BreadcrumbList',
      itemListElement: [
        {
          '@type': 'ListItem',
          position: 1,
          name: 'Home',
          item: Astro.site?.toString(),
        },
        {
          '@type': 'ListItem',
          position: 2,
          name: 'Use Cases',
          item: hubUrl,
        },
        {
          '@type': 'ListItem',
          position: 3,
          name: useCase.title,
          item: canonicalUrl,
        },
      ],
    };
    ---

    <Layout
      title={`${useCase.title} — QRCraft`}
      description={useCase.excerpt}
    >
      <SEO
        slot="head"
        title={`${useCase.title} — QRCraft`}
        description={useCase.excerpt}
        canonical={canonicalUrl}
        openGraph={{
          basic: {
            title: `${useCase.title} — QRCraft`,
            type: 'article',
            image: new URL('/og-image.png', Astro.site).toString(),
            url: canonicalUrl,
          },
          optional: {
            description: useCase.excerpt,
            siteName: 'QRCraft',
          },
        }}
      />
      <script type="application/ld+json" set:html={JSON.stringify(breadcrumbSchema)} slot="head" />

      <Header />
      <main>
        <article class="max-w-3xl mx-auto px-4 sm:px-6 lg:px-8 py-16">
          <!-- Breadcrumb -->
          <nav aria-label="Breadcrumb" class="mb-6">
            <ol class="flex items-center gap-2 text-sm text-gray-500 dark:text-slate-400" role="list">
              <li><a href="/" class="hover:text-[#2563EB] transition-colors">Home</a></li>
              <li aria-hidden="true">/</li>
              <li><a href="/use-cases/" class="hover:text-[#2563EB] transition-colors">Use Cases</a></li>
              <li aria-hidden="true">/</li>
              <li class="text-[#2563EB]" aria-current="page">{useCase.title}</li>
            </ol>
          </nav>

          <!-- Article header -->
          <h1 class="text-3xl sm:text-4xl font-bold text-gray-900 dark:text-white mb-4">
            {useCase.title}
          </h1>
          <p class="text-base text-gray-600 dark:text-slate-400 leading-relaxed mb-12">
            {useCase.excerpt}
          </p>

          <!-- Article body sections -->
          <div class="space-y-10 text-base text-gray-600 dark:text-slate-400 leading-relaxed">
            {useCase.body.map((section) => (
              <div>
                <h2 class="text-xl font-bold text-gray-900 dark:text-white mb-3">
                  {section.heading}
                </h2>
                {section.paragraphs.map((para) => (
                  <p class="mb-4">{para}</p>
                ))}
              </div>
            ))}
          </div>

          <!-- Bottom CTA -->
          <div class="mt-16 p-6 bg-gray-50 dark:bg-slate-800 rounded-xl border border-gray-200 dark:border-slate-700 text-center">

              Create your QR code
            </a>
          </div>
        </article>
      </main>
      <Footer />
