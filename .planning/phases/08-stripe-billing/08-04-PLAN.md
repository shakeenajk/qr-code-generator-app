  - Inline `<script>` (no type="module" needed — vanilla):
    ```js
    (function() {
      const monthlyBtn = document.getElementById('cta-monthly-btn');
      const annualBtn  = document.getElementById('cta-annual-btn');
      const upgradeBtn = document.getElementById('cta-upgrade-btn');

      const PRICES = {
        monthly: import.meta.env.PUBLIC_STRIPE_PRICE_PRO_MONTHLY,
        annual:  import.meta.env.PUBLIC_STRIPE_PRICE_PRO_ANNUAL,
      };
      let cadence = 'monthly';

      function setActive(selected) {
        cadence = selected;
        const activeClass = ['bg-[#2563EB]', 'text-white'];
        const inactiveClass = ['text-gray-500', 'dark:text-slate-400'];
        if (selected === 'monthly') {
          monthlyBtn.classList.add(...activeClass);
          monthlyBtn.classList.remove(...inactiveClass);
          annualBtn.classList.remove(...activeClass);
          annualBtn.classList.add(...inactiveClass);
        } else {
          annualBtn.classList.add(...activeClass);
          annualBtn.classList.remove(...inactiveClass);
          monthlyBtn.classList.remove(...activeClass);
          monthlyBtn.classList.add(...inactiveClass);
        }
      }

      monthlyBtn.addEventListener('click', () => setActive('monthly'));
      annualBtn.addEventListener('click', () => setActive('annual'));

      upgradeBtn.addEventListener('click', async () => {
        upgradeBtn.disabled = true;
        upgradeBtn.textContent = 'Loading...';
        try {
          const res = await fetch('/api/checkout/create', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ priceId: PRICES[cadence] }),
          });
          if (!res.ok) throw new Error('checkout failed');
          const { url } = await res.json();
          window.location.href = url;
        } catch {
          upgradeBtn.disabled = false;
          upgradeBtn.textContent = 'Upgrade to Pro';
        }
      });
    })();
    ```
