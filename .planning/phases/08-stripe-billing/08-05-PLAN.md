**Vanilla JS at bottom of page (single `<script>` block):**

```js
(function() {
  // Price IDs from env — PUBLIC_ prefix makes them available in static Astro pages
  const PRICES = {
    starter: {
      monthly: import.meta.env.PUBLIC_STRIPE_PRICE_STARTER_MONTHLY,
      annual:  import.meta.env.PUBLIC_STRIPE_PRICE_STARTER_ANNUAL,
    },
    pro: {
      monthly: import.meta.env.PUBLIC_STRIPE_PRICE_PRO_MONTHLY,
      annual:  import.meta.env.PUBLIC_STRIPE_PRICE_PRO_ANNUAL,
    },
  };

  let cadence = 'monthly';

  // Toggle UI
  const toggleMonthly = document.getElementById('toggle-monthly');
  const toggleAnnual  = document.getElementById('toggle-annual');

  function setToggle(selected) {
    cadence = selected;
    const activeClass = ['bg-[#2563EB]', 'text-white'];
    const inactiveClass = ['text-gray-500', 'dark:text-slate-400'];
    if (selected === 'monthly') {
      toggleMonthly.classList.add(...activeClass);
      toggleMonthly.classList.remove(...inactiveClass);
      toggleAnnual.classList.remove(...activeClass);
      toggleAnnual.classList.add(...inactiveClass);
    } else {
      toggleAnnual.classList.add(...activeClass);
      toggleAnnual.classList.remove(...inactiveClass);
      toggleMonthly.classList.remove(...activeClass);
      toggleMonthly.classList.add(...inactiveClass);
    }
    // Swap price spans
    document.querySelectorAll('.monthly-price').forEach(el => el.classList.toggle('hidden', selected !== 'monthly'));
    document.querySelectorAll('.annual-price').forEach(el => el.classList.toggle('hidden', selected !== 'annual'));
    document.querySelectorAll('.annual-savings').forEach(el => el.classList.toggle('hidden', selected !== 'annual'));
  }

  toggleMonthly.addEventListener('click', () => setToggle('monthly'));
  toggleAnnual.addEventListener('click',  () => setToggle('annual'));

  // Checkout
  async function startCheckout(tier, btn) {
    const priceId = PRICES[tier][cadence];
    btn.disabled = true;
    const originalText = btn.textContent;
    btn.textContent = 'Loading...';
    try {
      const res = await fetch('/api/checkout/create', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ priceId }),
      });
      if (res.status === 401) {
        // Not signed in — redirect to sign-in, then back to pricing
        window.location.href = '/sign-in?redirect=/pricing';
        return;
      }
      if (!res.ok) throw new Error('checkout failed');
      const { url } = await res.json();
      window.location.href = url;
    } catch {
      btn.disabled = false;
      btn.textContent = originalText;
    }
  }

  document.getElementById('starter-cta').addEventListener('click', function() {
    startCheckout('starter', this);
  });
  document.getElementById('pro-cta').addEventListener('click', function() {
    startCheckout('pro', this);
  });
})();
```
