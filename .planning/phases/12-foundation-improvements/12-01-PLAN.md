```typescript
// Centralized tier limits ‚Äî single source of truth for all enforcement logic.
// Import TIER_LIMITS wherever QR count checks are needed.

export type TierKey = 'free' | 'starter' | 'pro';

export interface TierLimitValues {
  /** Maximum total saved QR codes (static + dynamic). Infinity for unlimited. */
  totalQr: number;
  /** Maximum dynamic QR codes. Infinity for unlimited. */
  dynamicQr: number;
}

export const TIER_LIMITS: Record<TierKey, TierLimitValues> = {
  free:    { totalQr: 5,        dynamicQr: 3   },
  starter: { totalQr: 100,      dynamicQr: 10  },
  pro:     { totalQr: Infinity, dynamicQr: Infinity },
};
```

No default export. Named exports only.
