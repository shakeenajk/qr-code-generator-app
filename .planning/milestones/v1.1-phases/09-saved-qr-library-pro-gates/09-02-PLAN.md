<!-- Source: src/pages/api/subscription/status.ts (established auth pattern) -->
```typescript
export const prerender = false;
import type { APIRoute } from 'astro';
import { db } from '../../../db/index';
import { subscriptions } from '../../../db/schema';
import { eq } from 'drizzle-orm';

export const GET: APIRoute = async ({ locals }) => {
  const { userId } = locals.auth();
  if (!userId) return new Response('Unauthorized', { status: 401 });
  const sub = await db.query.subscriptions.findFirst({ where: eq(subscriptions.userId, userId) });
  const tier = sub?.tier ?? 'free';
  return new Response(JSON.stringify({ tier }), { status: 200, headers: { 'Content-Type': 'application/json' } });
};
```
