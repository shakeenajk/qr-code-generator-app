    **src/components/dashboard/MobileTabBar.astro:**
    ```astro
    ---
    import { LayoutGrid, BarChart2, Settings } from 'lucide-astro';

    interface Props {
      activeSection?: string;
    }

    const { activeSection = 'my-qr-codes' } = Astro.props;

    const tabs = [
      { href: '/dashboard', id: 'my-qr-codes', label: 'QR Codes', Icon: LayoutGrid },
      { href: '/dashboard/analytics', id: 'analytics', label: 'Analytics', Icon: BarChart2 },
      { href: '/dashboard/settings', id: 'settings', label: 'Settings', Icon: Settings },
    ];
    ---

    <nav
      class="fixed bottom-0 left-0 right-0 bg-white dark:bg-slate-900 border-t border-gray-200 dark:border-slate-700 flex md:hidden z-40 pb-safe"
      aria-label="Dashboard tabs"
    >
      {tabs.map(({ href, id, label, Icon }) => {
        const isActive = activeSection === id;
        return (
          <a
            href={href}
            class={`flex-1 flex flex-col items-center justify-center py-3 gap-0.5 text-xs font-medium transition-colors ${
              isActive
                ? 'text-blue-600 dark:text-blue-400'
                : 'text-gray-500 dark:text-slate-400'
            }`}
            aria-current={isActive ? 'page' : undefined}
          >
            <Icon class="w-5 h-5" />
            <span class="sr-only">{label}</span>
          </a>
        );
      })}
    </nav>
    ```

    Note: `pb-safe` uses the CSS env() safe-area-inset-bottom for iPhone notch support. Add this to the Tailwind config or use inline style if the class isn't defined:
    ```astro
    style="padding-bottom: env(safe-area-inset-bottom, 0)"
    ```
    Use the inline style approach since Tailwind v4 custom utilities may not be set up yet.

    **src/components/dashboard/DashboardLayout.astro:**
    ```astro
    ---
    import Header from '../Header.astro';
    import Sidebar from './Sidebar.astro';
    import MobileTabBar from './MobileTabBar.astro';

    interface Props {
      activeSection?: string;
      title?: string;
    }

    const { activeSection = 'my-qr-codes', title = 'Dashboard — QRCraft' } = Astro.props;
    ---

    <!doctype html>
    <html lang="en">
      <head>
        <meta charset="UTF-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1" />
        <title>{title}</title>
        <meta name="robots" content="noindex, nofollow" />
        <!-- Dashboard pages should not be indexed -->
        <link rel="stylesheet" href="/src/styles/global.css" />
      </head>
      <body class="bg-gray-50 text-gray-900 antialiased dark:bg-[#0f172a] dark:text-slate-100">
        <!-- Top header (full width) -->
        <Header />

        <!-- Main content area: sidebar + page content -->
        <div class="flex min-h-[calc(100vh-4rem)]">
          <Sidebar activeSection={activeSection} />

          <!-- Page content -->
          <main class="flex-1 overflow-auto pb-20 md:pb-0">
            <slot />
          </main>
        </div>

        <!-- Mobile tab bar (fixed at bottom, visible on mobile only) -->
        <MobileTabBar activeSection={activeSection} />
      </body>
    </html>
    ```

    Note: DashboardLayout is a standalone HTML document layout (like Layout.astro) because the dashboard has a fundamentally different structure (sidebar, no FAQ schema, noindex meta). It imports global.css directly. The `pb-20 md:pb-0` on `<main>` provides space for the mobile tab bar.
