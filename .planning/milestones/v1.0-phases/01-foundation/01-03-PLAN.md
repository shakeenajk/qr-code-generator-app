```astro
---
// src/layouts/Layout.astro
import '../styles/global.css';
import { FAQ_ITEMS } from '../data/faq';

interface Props {
  title: string;
  description: string;
  ogImage?: string;
}

const {
  title,
  description,
  ogImage = '/og-image.png',
} = Astro.props;

const canonicalUrl = new URL(Astro.url.pathname, Astro.site);
const ogImageUrl = new URL(ogImage, Astro.site);

const webAppSchema = {
  '@context': 'https://schema.org',
  '@type': 'WebApplication',
  name: 'QRCraft',
  url: Astro.site?.toString(),
  description:
    'Free online QR code generator. Create custom QR codes for URLs, text, WiFi, and contact cards — no signup required.',
  applicationCategory: 'UtilitiesApplication',
  operatingSystem: 'All',
  browserRequirements: 'Requires JavaScript.',
  offers: {
    '@type': 'Offer',
    price: '0',
    priceCurrency: 'USD',
  },
};

const faqSchema = {
  '@context': 'https://schema.org',
  '@type': 'FAQPage',
  mainEntity: FAQ_ITEMS.map((item) => ({
    '@type': 'Question',
    name: item.question,
    acceptedAnswer: {
      '@type': 'Answer',
      text: item.answer,
    },
  })),
};
---

<!doctype html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <title>{title}</title>
    <meta name="description" content={description} />
    <link rel="canonical" href={canonicalUrl} />

    <!-- Open Graph -->
    <meta property="og:type" content="website" />
    <meta property="og:url" content={canonicalUrl} />
    <meta property="og:title" content={title} />
    <meta property="og:description" content={description} />
    <meta property="og:image" content={ogImageUrl} />
    <meta property="og:image:width" content="1200" />
    <meta property="og:image:height" content="630" />
    <meta property="og:site_name" content="QRCraft" />

    <!-- Twitter/X Card -->
    <meta name="twitter:card" content="summary_large_image" />
    <meta name="twitter:title" content={title} />
    <meta name="twitter:description" content={description} />
    <meta name="twitter:image" content={ogImageUrl} />

    <!-- JSON-LD: WebApplication schema (SEO-03) -->
    <!-- MUST use set:html — Astro escapes JSON without it, producing invalid JSON-LD -->
    <script type="application/ld+json" set:html={JSON.stringify(webAppSchema)} />

    <!-- JSON-LD: FAQPage schema (SEO-04) — derived from same FAQ_ITEMS as visible FAQ section -->
    <script type="application/ld+json" set:html={JSON.stringify(faqSchema)} />

    <slot name="head" />
  </head>
  <body class="bg-white text-gray-900 antialiased">
    <slot />
  </body>
</html>
```

CRITICAL implementation notes:
- Both JSON-LD `<script>` tags MUST use `set:html={JSON.stringify(...)}` — never raw interpolation
- `FAQ_ITEMS` imported from `src/data/faq.ts` — same array as the visible FAQ component uses
- `bg-white text-gray-900` on `<body>` fulfills BRAND-02 (white background, dark text) per locked decisions
- `Astro.site` is available because `astro.config.mjs` sets `site: 'https://qrcraft.app'`
